project(ImageEffect LANGUAGES CXX)

find_package(Qt5 COMPONENTS Core Quick LinguistTools REQUIRED)
find_package(Qt5QuickCompiler)

qtquick_compiler_add_resources(RESOURCES ressources.qrc ${CMAKE_CURRENT_LIST_DIR}/src/qml/qml.qrc)

if(ANDROID)
    add_library(ImageEffect SHARED
      ${CMAKE_CURRENT_LIST_DIR}/src/main.cpp
      ${RESOURCES}
      ${TS_FILES}
    )
else()
    add_executable(ImageEffect
      ${CMAKE_CURRENT_LIST_DIR}/src/main.cpp
      ${RESOURCES}
      ${TS_FILES}
    )
endif()

target_compile_definitions(ImageEffect
  PRIVATE $<$<OR:$<CONFIG:Debug>,$<CONFIG:RelWithDebInfo>>:QT_QML_DEBUG>)
target_link_libraries(ImageEffect
  PRIVATE Qt5::Core Qt5::Quick)
